#regression analysis -->
x=c(151,174,138,186,179,163,152,131) #heighr in cm
y=c(63,81,56,91,76,72,62,48) #weight in kg

relation=lm(y~x)


a=data.frame(x=170)

result=predict(relation,a)

print(result)

png(file="regression.png")

plot(y,x,col="blue",main="height weight analysis" ,abline(lm(x~y)),pch=16,cex=1.3,xlab="Weight",ylab="height")

dev.off()
